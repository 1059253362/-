import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.tree import plot_tree
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.ensemble import RandomForestRegressor

df = pd.read_csv('Desktop/data_upadated.csv')

# Features and labels
X = df.drop(['price', 'Community_Name', 'price_per_sqm'], axis=1)
y = df['price']

# Data partitioning
X_train, X_val, y_train, y_val = train_test_split(X, y, test_size=0.3, random_state=42)

# Feature scaling
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_val_scaled = scaler.transform(X_val)

# Initialize the model
model = LinearRegression()

# Training model
model.fit(X_train_scaled, y_train)

# Model prediction
y_train_pred = model.predict(X_train_scaled)
y_val_pred = model.predict(X_val_scaled)

#Training set evaluation
train_mse = mean_squared_error(y_train, y_train_pred)
train_r2 = r2_score(y_train, y_train_pred)
train_mape = np.mean(np.abs((y_train - y_train_pred) / y_train)) * 100
    
#Validation set evaluation
val_mse = mean_squared_error(y_val, y_val_pred)
val_r2 = r2_score(y_val, y_val_pred)
val_mape = np.mean(np.abs((y_val - y_val_pred) / y_val)) * 100
    
print(f"Training Set - MSE: {train_mse}, R Score: {train_r2}, MAPE: {train_mape}%")
print(f"Validation Set - MSE: {val_mse}, R Score: {val_r2}, MAPE: {val_mape}%")
print("\nModel Parameters:")
print("Coefficients (weights):", model.coef_)
print("Intercept:", model.intercept_)

from sklearn.model_selection import RandomizedSearchCV
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error
from scipy.stats import randint
import numpy as np

def mean_absolute_percentage_error(y_true, y_pred): 
    y_true, y_pred = np.array(y_true), np.array(y_pred)
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

param_dist = {
    'n_estimators': randint(10, 200),
    'min_samples_split': randint(2, 10),
    'max_depth': randint(3, 20)
}

model = RandomForestRegressor(random_state=4)

scoring = {'neg_mean_absolute_percentage_error': 'neg_mean_absolute_percentage_error'}
random_search = RandomizedSearchCV(estimator=model, param_distributions=param_dist, n_iter=100, cv=5, scoring=scoring, refit='neg_mean_absolute_percentage_error')

random_search.fit(X_train_scaled, y_train)

print("Best parameters:", random_search.best_params_)
print("Best score (negative MAPE):", random_search.best_score_)

best_model = random_search.best_estimator_
y_val_pred = best_model.predict(X_val_scaled)

val_mape = mean_absolute_percentage_error(y_val, y_val_pred)
print("Validation Set - Mean Absolute Percentage Error:", val_mape)
